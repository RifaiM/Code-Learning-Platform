Introduction

Form validation is the process of checking user input to ensure it meets specific requirements before submission. JavaScript form validation provides immediate feedback to users, improving the user experience by catching errors before data is sent to the server.

üö´ Prevent Submission with preventDefault

event.preventDefault() stops the form from submitting when validation fails, keeping users on the page to fix errors.

// HTML: <form id="myForm" onsubmit="return validateForm(event)">
// JavaScript: Prevent submission on validation failure
function validateForm(event) {
    let input = document.getElementById("username");
    if (input.value.length === 0) {
        event.preventDefault(); // Stop form submission
        alert("Please fill in the username!");
        return false;
    }
    return true;
}

‚úÖ Good Form Validation Practice:

// Clear form structure and validation
<form id="registrationForm">
    <input type="text" id="username" placeholder="Username">
    <button type="submit">Register</button>
    <div id="errorMessage"></div>
</form>

// JavaScript with proper validation
document.getElementById("registrationForm").onsubmit = function(event) {
    let usernameInput = document.getElementById("username");
    let errorDiv = document.getElementById("errorMessage");
    
    // Clear previous errors
    errorDiv.innerHTML = "";
    
    // Validate username
    if (usernameInput.value.trim().length === 0) {
        errorDiv.innerHTML = "Username is required!";
        event.preventDefault();
        return false;
    }
    
    if (usernameInput.value.length < 3) {
        errorDiv.innerHTML = "Username must be at least 3 characters!";
        event.preventDefault();
        return false;
    }
    
    // Success
    errorDiv.innerHTML = "‚úÖ Username is valid!";
    errorDiv.style.color = "green";
    return true;
};
